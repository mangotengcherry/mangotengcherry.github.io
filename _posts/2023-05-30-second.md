```python
import random
def prepare_bag(total,hit):
    lst = [] 
    for i in range(hit):
        lst.append(1)
    for i in range(total - hit):
        lst.append(0)
    random.shuffle(lst)
    return lst
```

## The Hungry Dice-Player Problem


```python
def roll():
    from random import randint
    return randint(1,6)

def dice_game():
    strikes = 0
    winnings = 0
    while strikes < 3:
        die1 = roll()
        die2 = roll()
        if die1 == die2:
            strikes +=1
        else:
            winnings += die1 + die2
    return winnings

def average_winnings(runs):
    total = 0
    for n in range(runs):
        total = total + dice_game()
    return total / runs
```

```python
from random import shuffle
def clue_student(pairs, trials):
    num_correct = 0
    matching = list(range(pairs))
    for i in range(trials):
        shuffle(matching)
        for j in range(pairs):
            if matching[j] == j:
                num_correct = num_correct + 1
    return num_correct / trials
```

## Umbrella Quandary Problem


```python
from random import random
def q_umbrella(p) : 
    wet = False
    trips = 0
    location = 0
    umbrellas = [1,1]
    
    while (not wet):
        if random() < p:
            if umbrellas[location] == 0:
                wet = True
            else:
                trips += 1
                umbrellas[location] -= 1
                location = 1 - location
                umbrellas[location] += 1
                
        else:
            trips += 1
            location = 1 - location
    
    return trips
            
```
